# TO CHECK WHICH PROCESS IS EXECUTING IN BACKGROUND:
netstat -ano | findstr :5000
taskkill /PID 17664 /F 


# TO UNDO GIT ADD:
git restore --staged <file>

# TO UNDO GIT COMMIT:
git reset --hard HEAD~1


# TO DELETE MASTER BRANCH AND TO MOVE TO MAIN:
git branch -M main
git push -u origin main


# SUBMODULE TEMPLATE:
[submodule "design"]
	path = docs/design
	url = https://github.com/SENODROOM/QLang-DesginDocs.git
	branch = main

# USE GIT SUBTREE:
git remote add <remote-name> https://github.com/SENODROOM/Python.git
git subtree add --prefix=<folder-path> <remote-name> <branch>
git subtree pull --prefix=repo1 repo1 main
git subtree push --prefix=repo1 repo1 main

# TO UPDATE THE GITHUB SUBMODULES TO LATEST COMMITS:
git submodule update --remote --merge
git add frontend backend
git commit -m "Update submodules to latest commits"
git push



# TO CREATE A NPM PACKAGE:
npm init -y
npm search package-name
npm publish --access public


# TO UPDATE NPM PACKAGE:
---- For patch 
npm version patch 
---- For minor 
npm version minor 
---- For major 
npm version major
npm publish --access public

--- For project
npm update package-name
